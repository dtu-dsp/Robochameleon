152  function [fn, pn] = genNoise(obj)
153  % Time and frequency vectors (we double everything)
154  freqs = [-flipud(obj.model.FMfreq);0; obj.model.FMfreq];
155  time=linspace(-obj.model.Lpsd,obj.model.Lpsd,2*obj.model.Lpsd+1)/obj.Fs;
156  % Frequency domain filters
157  H_FN = obj.FMnoiseCal(:);
158  if length(H_FN)>1
159  H_FN = [ flipud(H_FN(:)); 0; H_FN(:)]/2;
160  H_FN = sqrt(H_FN);
161  % Time domain filters
162  h_fn = obj.idft(freqs, H_FN, time);
163  h_fn = real(h_fn(:));
164  else
165  h_fn = obj.Fs*sqrt(H_FN/2); %Hacked:
166  %std(diff(pn)) = sqrt(2*pi*linewidth/obj.Fs)
167  %H_FN = linewidth/pi
168  %algebra based on lines 308 to 310
169  %gives this equation
170  end
171  % Frequency and phase noise generation
172  noise = awgn(zeros(2*obj.Lnoise,1),0)/sqrt(obj.Fs);
173  fn =conv(noise,h_fn,'valid');
174  pn = cumsum(2*pi*fn/obj.Fs);
175  if obj.limitPn>0
176  limit=[-1 1]*obj.limitPn;
177  pn = pn-mean(pn);
178  pn(pn>max(limit)) = -pn(pn>max(limit))+2*max(limit);
179  pn(pn<min(limit)) = -pn(pn<min(limit))+2*min(limit);
180  end
181  obj.Lnoise = 2*(obj.Lnoise - obj.model.Lpsd);
182  end
