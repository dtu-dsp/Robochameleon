106  function [fn, pn] = genNoise(obj)
107  % Time and frequency vectors (we double everything)
108  freqs = [-flipud(obj.model.FMfreq);0; obj.model.FMfreq];
109  time=linspace(-obj.model.Lpsd,obj.model.Lpsd,2*obj.model.Lpsd+1)/obj.Fs;
110  % Frequency domain filters
111  H_FN = obj.FMnoiseCal(:);
112  H_FN = [ flipud(H_FN(:)); 0; H_FN(:)]/2;
113  H_FN = sqrt(H_FN);
114  % Time domain filters
115  h_fn = obj.idft(freqs, H_FN, time);
116  h_fn = real(h_fn(:));
117  % Frequency and phase noise generation
118  noise = awgn(zeros(2*obj.Lnoise,1),0)/sqrt(obj.Fs);
119  fn =conv(noise,h_fn,'valid');
120  pn = cumsum(2*pi*fn/obj.Fs);
121  if obj.limitPn>0
122  limit=[-1 1]*obj.limitPn;
123  pn = pn-mean(pn);
124  pn(pn>max(limit)) = -pn(pn>max(limit))+2*max(limit);
125  pn(pn<min(limit)) = -pn(pn<min(limit))+2*min(limit);
126  end
127  obj.Lnoise = 2*(obj.Lnoise - obj.model.Lpsd);
128  end
