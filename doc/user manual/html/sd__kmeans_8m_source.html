141 function d=DistMatrix(A,B)
142 %%%%%%%%%%%%%%%%%%%%%%%%%
143 % DISTMATRIX return distance matrix between points in A=[x1 y1 ... w1] and in B=[x2 y2 ... w2]
144 % Copyright (c) 2005 by Kardi Teknomo, http://people.revoledu.com/kardi/
145 %
146 % Numbers of rows (represent points) in A and B are not necessarily the same.
147 % It can be use for distance-in-a-slice (Spacing) or distance-between-slice (Headway),
148 %
149 % A and B must contain the same number of columns (represent variables of n dimensions),
150 % first column is the X coordinates, second column is the Y coordinates, and so on.
151 % The distance matrix is distance between points in A as rows
152 % and points in B as columns.
153 % example: Spacing= dist(A,A)
154 % Headway = dist(A,B), with hA ~= hB or hA=hB
155 % A=[1 2 3; 4 5 6; 2 4 6; 1 2 3]; B=[4 5 1; 6 2 0]
156 % dist(A,B)= [ 4.69 5.83;
157 % 5.00 7.00;
158 % 5.48 7.48;
159 % 4.69 5.83]
160 %
161 % dist(B,A)= [ 4.69 5.00 5.48 4.69;
162 % 5.83 7.00 7.48 5.83]
163 %%%%%%%%%%%%%%%%%%%%%%%%%%%
164 [hA,wA]=size(A);
165 [hB,wB]=size(B);
166 if wA ~= wB, error(' second dimension of A and B must be the same'); end
167 for k=1:wA
168  C{k}= repmat(A(:,k),1,hB);
169  D{k}= repmat(B(:,k),1,hA);
170 end
171 S=zeros(hA,hB);
172 for k=1:wA
173  S=S+(C{k}-D{k}').^2;
174 end
175 d=sqrt(S);
