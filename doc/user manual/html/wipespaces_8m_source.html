wipespaces.m
1 % To remove a Matlab trailing whitespace in the editor
2 % Original Author: Sam Roberts
3 % Improved by: Simone Gaiarin <simgunz@gmail.com>
4 % http://stackoverflow.com/questions/19770347/how-to-auto-remove-trailing-whitespaces-on-save-in-matlab
5 % Modified by Mark Harfouche to remember cursor location
6 %
7 %
8 % Temp variable for shortcut. Give it an unusual name so it's unlikely to
9 % conflict with anything in the workspace.
10 shtcutwh__ = struct;
11 
12 % Check that the editor is available.
13 if ~matlab.desktop.editor.isEditorAvailable
14  return
15 end
16 
17 % Check that a document exists.
18 shtcutwh__.activeDoc = matlab.desktop.editor.getActive;
19 if isempty(shtcutwh__.activeDoc)
20  return
21 end
22 
23 % save the old cursor location
24 shtcutwh__.Selection = shtcutwh__.activeDoc.Selection;
25 
26 % Get the current text.
27 shtcutwh__.txt = shtcutwh__.activeDoc.Text;
28 
29 % Remove trailing whitespaces from each line.
30 shtcutwh__.allLines = regexp(shtcutwh__.txt, '[^\n]*(\n)|[^\n]*$', 'match');
31 shtcutwh__.lines = deblank(shtcutwh__.allLines);
32 
33 % Uncomment this block to remove trailing whitespaces only from each non-empty line.
34 % This method preserve the indentation whitespaces inserted by matlab in the empty lines.
35 emptyLinesIdx = cellfun(@isempty, shtcutwh__.lines);
36 onlySpacesLinesIdx = cellfun(@(x) length(x)>1, shtcutwh__.allLines);
37 idx = emptyLinesIdx & onlySpacesLinesIdx;
38 if any(onlySpacesLinesIdx)
39  shtcutwh__.lines(idx) = regexprep(shtcutwh__.allLines(idx), '\n', '');
40 end
41 
42 % remove the trailing blank lines
43 for n = length(shtcutwh__.lines):-1:1
44  if isempty(regexp(shtcutwh__.lines{n}, '.*\S.*', 'once'))
45  shtcutwh__.lines(n) = [];
46  else
47  break
48  end
49 end
50 
51 % Reconcatenate lines.
52 shtcutwh__.addNewline = @(x)sprintf('%s\n',x);
53 
54 shtcutwh__.lines = cellfun(shtcutwh__.addNewline, shtcutwh__.lines, 'UniformOutput', false);
55 
56 % If you always want to add a newline at the end of the file, comment this line out
57 % Remove the last newline character
58 % shtcutwh__.lines{end}(end) = '';
59 
60 shtcutwh__.newtxt = horzcat(shtcutwh__.lines{:});
61 
62 % Set the current text.
63 shtcutwh__.activeDoc.Text = shtcutwh__.newtxt;
64 
65 % Place the cursor back
66 shtcutwh__.activeDoc.Selection = shtcutwh__.Selection;
67 
68 % Delete temp variable.
69 clear shtcutwh__
